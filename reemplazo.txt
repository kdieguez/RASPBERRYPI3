FOR EACH ROW
DECLARE v_cnt NUMBER;
BEGIN
  SELECT COUNT(*) INTO v_cnt
    FROM AEROLINEA3.RESERVA r
    JOIN AEROLINEA3.RESERVA_ITEM ri ON ri.ID_RESERVA = r.ID_RESERVA
   WHERE r.ID_USUARIO = :NEW.ID_USUARIO_AUTOR
     AND ri.ID_VUELO   = :NEW.ID_VUELO
     AND r.ID_ESTADO   IN (2,3); -- CONFIRMADA/PAGADA (ajusta a tu flujo)

  IF v_cnt = 0 THEN
    RAISE_APPLICATION_ERROR(-20033,'Solo usuarios con reserva pueden calificar.');
  END IF;
END;

--  DDL for Procedure PR_CANCELAR_RESERVA
--------------------------------------------------------
set define off;

  CREATE OR REPLACE EDITIONABLE PROCEDURE "AEROLINEA3"."PR_CANCELAR_RESERVA" (
  p_id_usuario IN AEROLINEA3.USUARIO.ID_USUARIO%TYPE,
  p_id_reserva IN AEROLINEA3.RESERVA.ID_RESERVA%TYPE
) AS
  v_cnt NUMBER;
BEGIN
  -- Validar que sea suya y esté en estado 1 (PAGADA)
  SELECT COUNT(*) INTO v_cnt
    FROM AEROLINEA3.RESERVA
   WHERE ID_RESERVA = p_id_reserva
     AND ID_USUARIO = p_id_usuario
     AND ID_ESTADO  = 1;

  IF v_cnt = 0 THEN
    RAISE_APPLICATION_ERROR(-20005, 'Reserva inválida o no cancelable.');
  END IF;

  -- Reponer cupos
  FOR rec IN (
    SELECT ID_VUELO, ID_CLASE
      FROM AEROLINEA3.RESERVA_ITEM
     WHERE ID_RESERVA = p_id_reserva
  ) LOOP
    AEROLINEA3.PR_REPONER_CUPO(rec.ID_VUELO, rec.ID_CLASE, 1);
  END LOOP;

  -- Marcar CANCELADA (2)
  UPDATE AEROLINEA3.RESERVA
     SET ID_ESTADO = 2
   WHERE ID_RESERVA = p_id_reserva;

  UPDATE AEROLINEA3.RESERVA_ITEM
     SET ID_ESTADO_RESERVA = 2
   WHERE ID_RESERVA = p_id_reserva;
END;

--------------------------------------------------------
--  DDL for Procedure PR_CHECKOUT_CARRITO
--------------------------------------------------------
set define off;

  CREATE OR REPLACE EDITIONABLE PROCEDURE "AEROLINEA3"."PR_CHECKOUT_CARRITO" (
  p_id_usuario  IN  AEROLINEA3.USUARIO.ID_USUARIO%TYPE,
  p_id_carrito  IN  AEROLINEA3.CARRITO.ID_CARRITO%TYPE,
  p_reserva_out OUT AEROLINEA3.RESERVA.ID_RESERVA%TYPE
) AS
  v_total   NUMBER(12,2);
  v_dummy   NUMBER;
BEGIN
  -- Validar carrito del usuario
  BEGIN
    SELECT 1 INTO v_dummy
      FROM AEROLINEA3.CARRITO
     WHERE ID_CARRITO = p_id_carrito
       AND ID_USUARIO = p_id_usuario;
  EXCEPTION WHEN NO_DATA_FOUND THEN
    RAISE_APPLICATION_ERROR(-20002,'Carrito inválido.');
  END;

  -- Total
  SELECT AEROLINEA3.FN_TOTAL_CARRITO(p_id_carrito) INTO v_total FROM dual;
  IF v_total <= 0 THEN
    RAISE_APPLICATION_ERROR(-20003,'Carrito vacío.');
  END IF;

  -- Descontar cupos
  FOR rec IN (
    SELECT ID_VUELO, ID_CLASE, CANTIDAD
      FROM AEROLINEA3.CARRITO_ITEM
     WHERE ID_CARRITO = p_id_carrito
  ) LOOP
    IF AEROLINEA3.FN_DESCONTAR_CUPO(rec.ID_VUELO, rec.ID_CLASE, rec.CANTIDAD) = 0 THEN
      RAISE_APPLICATION_ERROR(-20004,'Sin cupo para vuelo '||rec.ID_VUELO||' clase '||rec.ID_CLASE);
    END IF;
  END LOOP;

  -- Crear RESERVA en estado 1 (PAGADA)
  INSERT INTO AEROLINEA3.RESERVA(ID_USUARIO, ID_ESTADO, TOTAL)
  VALUES (p_id_usuario, 1, v_total)
  RETURNING ID_RESERVA INTO p_reserva_out;

  -- Pasar items a RESERVA_ITEM en estado 1
  INSERT INTO AEROLINEA3.RESERVA_ITEM
      (ID_RESERVA, ID_VUELO, ID_CLASE, PRECIO_UNITARIO, ID_ESTADO_RESERVA)
  SELECT p_reserva_out, ID_VUELO, ID_CLASE, PRECIO_UNITARIO, 1
    FROM (
      SELECT ci.ID_VUELO, ci.ID_CLASE, ci.PRECIO_UNITARIO, LEVEL AS n
        FROM AEROLINEA3.CARRITO_ITEM ci
       WHERE ci.ID_CARRITO = p_id_carrito
  CONNECT BY LEVEL <= ci.CANTIDAD
         AND PRIOR ci.ID_ITEM = ci.ID_ITEM
         AND PRIOR SYS_GUID() IS NOT NULL
    );

  -- Vaciar carrito
  DELETE FROM AEROLINEA3.CARRITO_ITEM WHERE ID_CARRITO = p_id_carrito;
END;

  CREATE OR REPLACE EDITIONABLE PROCEDURE "AEROLINEA3"."PR_REPONER_CUPO" (
  p_id_vuelo IN SALIDA_CLASE.ID_VUELO%TYPE,
  p_id_clase IN SALIDA_CLASE.ID_CLASE%TYPE,
  p_cantidad IN NUMBER
) IS
BEGIN
  UPDATE SALIDA_CLASE
     SET CUPO_TOTAL = CUPO_TOTAL + p_cantidad
   WHERE ID_VUELO = p_id_vuelo
     AND ID_CLASE = p_id_clase;
END;

--------------------------------------------------------
--  DDL for Function FN_DESCONTAR_CUPO
--------------------------------------------------------

  CREATE OR REPLACE EDITIONABLE FUNCTION "AEROLINEA3"."FN_DESCONTAR_CUPO" (
  p_id_vuelo  IN SALIDA_CLASE.ID_VUELO%TYPE,
  p_id_clase  IN SALIDA_CLASE.ID_CLASE%TYPE,
  p_cantidad  IN NUMBER
) RETURN NUMBER
IS
BEGIN
  UPDATE SALIDA_CLASE
     SET CUPO_TOTAL = CUPO_TOTAL - p_cantidad
   WHERE ID_VUELO   = p_id_vuelo
     AND ID_CLASE   = p_id_clase
     AND CUPO_TOTAL >= p_cantidad;

  IF SQL%ROWCOUNT = 1 THEN
    RETURN 1;            -- Se descontó
  ELSE
    RETURN 0;            -- No había cupo suficiente
  END IF;
END;

  CREATE OR REPLACE EDITIONABLE FUNCTION "AEROLINEA3"."FN_GEN_CODIGO_RESERVA" 
RETURN VARCHAR2 IS
  v_code VARCHAR2(30);
BEGIN
  v_code := 'AR' || TO_CHAR(SYSDATE,'YYMMDD') || '-' ||
            LPAD(TO_CHAR(AEROLINEA3.SEQ_RESERVA_COD.NEXTVAL), 6, '0');
  RETURN v_code;
END;

--  DDL for Function FN_TOTAL_CARRITO
--------------------------------------------------------

  CREATE OR REPLACE EDITIONABLE FUNCTION "AEROLINEA3"."FN_TOTAL_CARRITO" (p_id_carrito IN NUMBER)
RETURN NUMBER IS v_total NUMBER(12,2);
BEGIN
  SELECT NVL(SUM(CANTIDAD * PRECIO_UNITARIO),0)
    INTO v_total
    FROM AEROLINEA3.CARRITO_ITEM
   WHERE ID_CARRITO = p_id_carrito;
  RETURN v_total;
END;
